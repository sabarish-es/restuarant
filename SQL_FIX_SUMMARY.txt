┌────────────────────────────────────────────────────────────────────────────┐
│ THE PROBLEM                                                                │
└────────────────────────────────────────────────────────────────────────────┘

Error Message:
  ❌ Unknown column 'key_name' in 'field list'

Root Cause:
  Two database initialization scripts had conflicting column names:
  
  ❌ initDatabase.js used:
     - Column: key_name
     - Column: key_value
     - Description: TEXT
  
  ✅ database.sql used:
     - Column: setting_key
     - Column: setting_value
     (No description field)

┌────────────────────────────────────────────────────────────────────────────┐
│ THE FIX                                                                    │
└────────────────────────────────────────────────────────────────────────────┘

File Modified: backend/scripts/initDatabase.js

BEFORE (Lines 126-140):
────────────────────────
-- Create settings table
CREATE TABLE IF NOT EXISTS settings (
  id INT AUTO_INCREMENT PRIMARY KEY,
  key_name VARCHAR(100) UNIQUE NOT NULL,        ❌ WRONG
  key_value TEXT,                               ❌ WRONG
  description TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  INDEX idx_key (key_name)                      ❌ WRONG
);

-- Insert default settings
INSERT IGNORE INTO settings (key_name, key_value, description) VALUES  ❌ WRONG
('restaurant_name', 'FoodHub Restaurant', 'Restaurant name'),
('restaurant_phone', '(123) 456-7890', 'Restaurant phone number'),
('restaurant_email', 'info@foodhub.com', 'Restaurant email'),
('tax_percentage', '5', 'Default tax percentage'),
('currency', 'USD', 'Currency symbol');


AFTER (Lines 126-138):
─────────────────────
-- Create settings table
CREATE TABLE IF NOT EXISTS settings (
  id INT AUTO_INCREMENT PRIMARY KEY,
  setting_key VARCHAR(100) UNIQUE NOT NULL,    ✅ CORRECT
  setting_value TEXT,                          ✅ CORRECT
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  INDEX idx_key (setting_key)                  ✅ CORRECT
);

-- Insert default settings
INSERT IGNORE INTO settings (setting_key, setting_value) VALUES  ✅ CORRECT
('restaurant_name', 'My Restaurant'),

('tax_rate', '0.05');
